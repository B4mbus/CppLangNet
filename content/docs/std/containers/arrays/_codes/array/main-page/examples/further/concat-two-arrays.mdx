```cpp title="Concat two arrays (C++20)"
#include <iostream>
#include <array>
#include <algorithm>

template <typename T, size_t N1, size_t N2>
auto concat(
		std::array<T, N1> const& lhs,
		std::array<T, N2> const& rhs
	)
{
	namespace rg = std::ranges;

	std::array<T, N1 + N2> result;

	rg::copy(lhs, result.begin());
	rg::copy(rhs, result.begin() + lhs.size());

	return result;
}

int main()
{
	auto left	= std::to_array({1, 2, 3});
	auto right	= std::to_array({4, 5, 6});
	
	auto both	= concat(left, right);

	for (int elem : both)
		std::cout << elem << ' ';
}
```
